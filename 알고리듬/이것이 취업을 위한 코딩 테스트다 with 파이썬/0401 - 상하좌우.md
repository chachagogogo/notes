파이썬 함수 내에서 전역변수를 참조하는 법을 몰라서 조금 헤맸다.

```python
n = int(input())
order = input().split()

row, column = 1, 1

def go_right():
	global column
	if column != n:
		column += 1

def go_left():
	global column
	if column != 1:
		column -= 1

def go_up():
	global row
	if row != 1:
		row -= 1

def go_down():
	global row
	if row != n:
		row += 1

for i in order:
	if i == 'R':
		go_right()
	if i == 'L':
		go_left()
	if i == 'U':
		go_up()
	if i == 'D':
		go_down()

print(row, column)
```
